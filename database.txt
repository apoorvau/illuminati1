CREATE TABLE CUSTOMER_DETAILS (CUST_ID INT PRIMARY KEY, CUST_NAME VARCHAR(50) NOT NULL, EMAILID VARCHAR(10) NOT NULL UNIQUE, PASSWORD VARCHAR(10) NOT NULL, ADDRESS VARCHAR(500), POSTAL_PIN INT NOT NULL);

CREATE TABLE ORDER_DETAILS (CUST_ID INT FOREIGN KEY, ORDER_ID INT PRIMARY KEY, ORDER_NAME VARCHAR(50) NOT NULL, QUANTITY_ORDER INT NOT NULL, STATUS VARCHAR(20) NOT NULL, STORE_ID INT NOT NULL);

CREATE TABLE STORE_DETAILS (STORE_ID INT PRIMARY KEY, STORE_NAME VARCHAR(50) NOT NULL, CONTACT_NO INT);

CREATE TABLE INVENTORY DETAILS (STORE_ID NUMBER(10), PRODUCT_ID NUMBER(10), AVAILABLE_STOCK NUMBER(10));

CREATE TABLE PRODUCT_DETAILS (PRODUCT_ID INT, PRODUCT_NAME VARCHAR(50), PRICE INT);
